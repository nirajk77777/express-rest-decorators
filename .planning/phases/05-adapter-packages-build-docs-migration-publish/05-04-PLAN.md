    **Step 2 — Write `.github/workflows/ci.yml`.** Use the Write tool. Content EXACTLY:

    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    concurrency:
      group: ci-${{ github.workflow }}-${{ github.ref }}
      cancel-in-progress: true

    jobs:
      quality:
        name: Node ${{ matrix.node }} / pool=${{ matrix.pool }}
        runs-on: ubuntu-latest
        strategy:
          fail-fast: false
          matrix:
            node: [20, 22, 24]
            pool: [forks, threads]
        steps:
          - uses: actions/checkout@v4

          - name: Set up pnpm
            uses: pnpm/action-setup@v6
            with:
              version: 10

          - name: Set up Node.js ${{ matrix.node }}
            uses: actions/setup-node@v4
            with:
              node-version: ${{ matrix.node }}
              cache: 'pnpm'

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Single-package invariant (BUILD-06)
            run: bash scripts/check-single-package.sh

          - name: Typecheck
            run: pnpm typecheck

          - name: Lint (Biome)
            run: pnpm lint

          - name: Run tests (pool=${{ matrix.pool }})
            run: pnpm test:${{ matrix.pool }}

          - name: Build (tshy dual ESM+CJS)
            run: pnpm build

          - name: Build smoke (BUILD-01 + emitDecoratorMetadata)
            run: pnpm test:build

          - name: Type resolution gate (attw, BUILD-07)
            run: pnpm exec attw --pack . --profile node16

          - name: Package config gate (publint, BUILD-07)
            run: pnpm exec publint
    ```

    Critical points:
    - `fail-fast: false` — when one cell fails we still want to see results from the others (distinguishes Node-version bugs from pool-isolation bugs).
    - `concurrency.cancel-in-progress: true` — superseded PRs don't burn minutes.
    - No `permissions:` block — ci.yml does NOT publish; release.yml (Plan 05-06) is where `id-token: write` lives.
    - The build smoke step (`pnpm test:build`) runs AFTER `pnpm build` in every cell. This means the emitDecoratorMetadata smoke is enforced on every Node version + pool combination — the load-bearing assertion runs 6 times per PR.
    - If Plan 05-03's findings forced `--ignore-rules false-cjs`, the package.json `prepublishOnly` and `release` scripts already reflect that. The bare `pnpm exec attw --pack . --profile node16` command in CI uses the same flag set as the local invocation only if the SUMMARY from 05-03 says no ignore was needed. **If 05-03's SUMMARY says `--ignore-rules false-cjs` was added**: change the attw step here to `pnpm exec attw --pack . --profile node16 --ignore-rules false-cjs` so CI matches local. Read `05-03-SUMMARY.md` before writing this file to decide.
    - Tests run BEFORE build because the test suite operates on `src/` (vitest + unplugin-swc), not on `dist/`. The smoke tests that depend on `dist/` are gated to `tests/build/**` and run after `pnpm build`.
